115  ArrayOf ArrayOf::integerRangeConstructor(indexType minval, indexType stepsize, indexType maxval, bool vert)
116  {
117  Dimensions Cdim;
118  if (stepsize == 0)
119  {
120  Cdim[0] = 1;
121  Cdim[1] = 0;
122 #ifdef NLS_INDEX_TYPE_64
123  return ArrayOf(NLS_INT32, Cdim, NULL, false);
124 #else
125  return ArrayOf(NLS_INT64, Cdim, NULL, false);
126 #endif
127  }
128  if (minval < maxval)
129  {
130  if (stepsize < 0)
131  {
132  Cdim[0] = 1;
133  Cdim[1] = 0;
134 #ifdef NLS_INDEX_TYPE_64
135  return ArrayOf(NLS_INT64, Cdim, NULL, false);
136 #else
137  return ArrayOf(NLS_INT32, Cdim, NULL, false);
138 #endif
139  }
140  }
141  if (minval > maxval)
142  {
143  if (stepsize > 0)
144  {
145  Cdim[0] = 1;
146  Cdim[1] = 0;
147 #ifdef NLS_INDEX_TYPE_64
148  return ArrayOf(NLS_INT64, Cdim, NULL, false);
149 #else
150  return ArrayOf(NLS_INT32, Cdim, NULL, false);
151 #endif
152  }
153  }
154  double dn = (double)((((maxval - minval) / stepsize) + 1));
155 #ifdef NLS_INDEX_TYPE_64
156  int64 n = (int64)floor((double)(dn));
157 #else
158  int32 n = (int32)floor((double)(dn));
159 #endif
160  if (vert)
161  {
162  Cdim[0] = n;
163  Cdim[1] = 1;
164  }
165  else
166  {
167  Cdim[0] = 1;
168  Cdim[1] = n;
169  }
170 #ifdef NLS_INDEX_TYPE_64
171  int64 *rp = (int64 *)allocateArrayOf(NLS_INT64, n);
172 #else
173  int32 *rp = (int32 *)allocateArrayOf(NLS_INT32, n);
174 #endif
175  if (dn == (double(n)))
176  {
177 #ifdef NLS_INDEX_TYPE_64
178  Eigen::Map<Eigen::Matrix<int64, Eigen::Dynamic, 1>> Range(rp, n);
179  Range = Eigen::Matrix<int64, Eigen::Dynamic, 1>::LinSpaced(n, minval, maxval);
180 #else
181  Eigen::Map<Eigen::VectorXi> Range(rp, n);
182  Range = Eigen::VectorXi::LinSpaced(n, minval, maxval);
183 #endif
184  }
185  else
186  {
187  // We must use another algo. in this case
188  // 1:2:10
189  indexType i = 0;
190  indexType v = minval;
191  while ((minval < maxval && v <= maxval) || (minval > maxval && v >= maxval))
192  {
193  rp[i] = v;
194  v = v + stepsize;
195  i++;
196  }
197  }
198 #ifdef NLS_INDEX_TYPE_64
199  return ArrayOf(NLS_INT64, Cdim, rp);
200 #else
201  return ArrayOf(NLS_INT32, Cdim, rp);
202 #endif
203  }
