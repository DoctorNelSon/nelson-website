176  indexType Dimensions::mapPoint(const Dimensions& point) throw (Exception)
177  {
178  indexType retval;
179  indexType nextCoeff;
180  indexType testableDims;
181  retval = 0;
182  nextCoeff = 1;
183  testableDims = (point.length < length) ? point.length : length;
184  for (indexType i = 0; i < testableDims; i++)
185  {
186  if ((point.data[i] < 0) || (point.data[i] >= data[i]))
187  {
188  throw Exception(_W("Index exceeds dimensions."));
189  }
190  retval += nextCoeff * point.data[i];
191  nextCoeff *= data[i];
192  }
193  for (sizeType j = testableDims; j < point.length; j++)
194  {
195  if (point.data[j] != 0)
196  {
197  throw Exception(_W("Index exceeds dimensions."));
198  }
199  }
200  return retval;
201  }
