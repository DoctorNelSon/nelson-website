28 {

66  }
67  }

69  default:

72  }
73  break;
74  }
75  if (cal == nullptr)
76  {
77  Error(eval, _W("Calendar not initialized."));
78  }
79  if (cal->getYear() < 1400 || cal->getYear() > 9999)
80  {
81  Error(eval, _W("Year value is wrong [1400, 9999] expected."));
82  }
83  if (!(cal->getMonth() > 0 && cal->getMonth() < 13))
84  {
85  Error(eval, _W("Month value is wrong [1, 12] expected."));
86  }

111  }
112  break;
113  default:
114  {
115  Error(eval, ERROR_WRONG_NUMBERS_OUTPUT_ARGS);
116  }
117  break;
118  }

120 }
