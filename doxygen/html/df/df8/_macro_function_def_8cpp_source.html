155 ArrayOfVector MacroFunctionDef::evaluateFunction(Evaluator *eval, ArrayOfVector& inputs, int nargout)
156 {
157  ArrayOfVector outputs;
158  Context* context = nullptr;
159  ArrayOf a;
160  size_t minCount = 0;
161  context = eval->getContext();
162  context->pushScope(name);
163  eval->pushDebug(wstring_to_utf8(fileName), name);
164  // Push our local functions onto the function scope
165  MacroFunctionDef *cp;
166  // Walk up until we get to the head of the list
167  cp = this;
168  while (cp->prevFunction != nullptr)
169  {
170  cp = cp->prevFunction;
171  }
172  cp = cp->nextFunction;
173  while (cp != nullptr)
174  {
175  context->insertMacroFunctionLocally((FuncPtr)cp);
176  cp = cp->nextFunction;
177  }
178  // When the function is called, the number of inputs is
179  // at sometimes less than the number of arguments requested.
180  // Check the argument count. If this is a non-varargin
181  // argument function, then use the following logic:
182  minCount = 0;
183  if (inputArgCount() != -1)
184  {
185  minCount = (inputs.size() < arguments.size()) ?
186  inputs.size() : arguments.size();
187  for (size_t i = 0; i<minCount; i++)
188  {
189  std::string arg(arguments[i]);
190  if (arg[0] == '&')
191  {
192  arg.erase(0, 1);
193  }
194  context->insertVariableLocally(arg, inputs[i]);
195  }
196  //context->insertVariableLocally("nargin", ArrayOf::doubleConstructor((double)minCount));
197  context->getCurrentScope()->setNargIn((int)minCount);
198  }
199  else
200  {
201  // Count the number of supplied arguments
202  size_t inputCount = inputs.size();
203  //context->insertVariableLocally("nargin", ArrayOf::doubleConstructor((double)inputCount));
204  context->getCurrentScope()->setNargIn((int)inputCount);
205  // Get the number of explicit arguments
206  int explicitCount = (int)arguments.size() - 1;
207  // For each explicit argument (that we have an input for),
208  // insert it into the scope.
209  minCount = (explicitCount < (int)inputCount) ? (size_t)explicitCount : (size_t)inputCount;
210  size_t i;
211  for (i = 0; i<minCount; i++)
212  {
213  std::string arg(arguments[i]);
214  if (arg[0] == '&')
215  {
216  arg.erase(0, 1);
217  }
218  context->insertVariableLocally(arg, inputs[i]);
219  }
220  inputCount -= minCount;
221  // Put minCount...inputCount
222  ArrayOf varg(NLS_CELL_ARRAY);
223  varg.vectorResize(inputCount);
224  ArrayOf* dp = (ArrayOf *)varg.getReadWriteDataPointer();
225  for (i = 0; i<inputCount; i++)
226  {
227  dp[i] = inputs[i + minCount];
228  }
229  context->insertVariableLocally("varargin", varg);
230  }
231  //context->insertVariableLocally("nargout", ArrayOf::doubleConstructor(nargout));
232  context->getCurrentScope()->setNargOut(nargout);
233  try
234  {
235  eval->block(code);
236  State state(eval->getState());
237  if (state < NLS_STATE_QUIT)
238  {
239  eval->resetState();
240  }
241  if (state == NLS_STATE_ABORT)
242  {
243  return scalarArrayOfToArrayOfVector(ArrayOf::emptyConstructor());
244  }
245  bool warningIssued = false;
246  if (outputArgCount() != -1)
247  {
248  outputs = ArrayOfVector(returnVals.size());
249  for (size_t i = 0; i<returnVals.size(); i++)
250  {
251  if (!context->lookupVariableLocally(returnVals[i], a))
252  {
253  if (!warningIssued)
254  {
255  eval->getInterface()->warningMessage(_W("one or more outputs not assigned in call."));
256  warningIssued = true;
257  }
258  a = ArrayOf::emptyConstructor();
259  }
260  outputs[i] = a;
261  }
262  }
263  else
264  {
265  outputs = ArrayOfVector(nargout);
266  int explicitCount = (int)returnVals.size() - 1;
267  // For each explicit argument (that we have), insert it
268  // into the scope.
269  for (int i = 0; i<explicitCount; i++)
270  {
271  if (!context->lookupVariableLocally(returnVals[i], a))
272  {
273  if (!warningIssued)
274  {
275  eval->getInterface()->warningMessage(_W("one or more outputs not assigned in call"));
276  warningIssued = true;
277  }
278  a = ArrayOf::emptyConstructor();
279  }
280  outputs[i] = a;
281  }
282  // Are there any outputs not yet filled?
283  if (nargout > explicitCount)
284  {
285  ArrayOf varargout;
286  // Yes, get a pointer to the "vargout" variable that should be defined
287  if (!context->lookupVariableLocally("varargout", varargout))
288  {
289  throw Exception(_W("The special variable 'varargout' was not defined as expected"));
290  }
291  if (varargout.getDataClass() != NLS_CELL_ARRAY)
292  {
293  throw Exception(_W("The special variable 'varargout' was not defined as a cell-array"));
294  }
295  // Get the data pointer
296  const ArrayOf *dp = ((const ArrayOf*)varargout.getDataPointer());
297  // Get the length
298  indexType varlen = varargout.getLength();
299  int toFill = nargout - explicitCount;
300  if ((double)toFill > (double)varlen)
301  {
302  throw Exception(_W("Not enough outputs in varargout to satisfy call"));
303  }
304  for (int i = 0; i<toFill; i++)
305  {
306  outputs[explicitCount + i] = dp[i];
307  }
308  }
309  }
310  // Check for arguments that were passed by reference, and
311  // update their values.
312  for (size_t i = 0; i<minCount; i++)
313  {
314  std::string arg(arguments[i]);
315  if (arg[0] == '&')
316  {
317  arg.erase(0, 1);
318  }
319  context->lookupVariableLocally(arg, inputs[i]);
320  }
321  context->popScope();
322  eval->popDebug();
323  return outputs;
324  }
325  catch (Exception& e)
326  {
327  context->popScope();
328  eval->popDebug();
329  e.what();
330  throw;
331  }
332 }

335 {

353 {

366  t->nextFunction->prevFunction = t;

372  }
