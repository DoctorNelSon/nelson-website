36 std::wstring BsdTerminal::getTextLine(std::wstring prompt, bool bIsInput)
37 {
38  if (prompt != L"")
39  {
40  fprintf(stdout, "%s", "\n");
41  this->diary.writeMessage(L"\n");
42  this->diary.writeMessage(prompt);
43  }
44  char *line = linenoise(wstring_to_utf8(prompt).c_str());
45  std::wstring retLineW = L"";
46  if (line)
47  {
48  std::string retLine = line;
49  linenoiseFree(line);
50  line = nullptr;
51  retLineW = utf8_to_wstring(retLine);
52  if (retLineW.empty())
53  {
54  retLineW = L"\n";
55  }
56  if (!bIsInput)
57  {
58  Nelson::History::addLine(retLineW);
59  }
60  this->diary.writeMessage(retLineW);
61  }
62  if (bIsInput)
63  {
64  if (boost::algorithm::ends_with(retLineW, L"\n"))
65  {
66  retLineW.pop_back();
67  }
68  Nelson::History::setToken(L"");
69  }
70  return retLineW;
71 }
