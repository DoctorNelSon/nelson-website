24 ArrayOfVector Nelson::DataStructuresGateway::struct2cellBuiltin(Evaluator* eval, int nLhs, const ArrayOfVector& argIn)
25 {
26  ArrayOfVector ret;
27  if (nLhs > 1)
28  {
29  Error(eval, ERROR_WRONG_NUMBERS_OUTPUT_ARGS);
30  }
31  if (argIn.size() != 1)
32  {
33  Error(eval, ERROR_WRONG_NUMBERS_INPUT_ARGS);
34  }
35  ArrayOf param1 = argIn[0];
36  if (!param1.isStruct())
37  {
38  Error(eval, ERROR_WRONG_ARGUMENT_1_TYPE_STRUCT_EXPECTED);
39  }
40  stringVector fieldnames = param1.getFieldNames();
41  size_t nbFields = fieldnames.size();
42  ArrayOf *elements = nullptr;
43  Dimensions dimsStruct = param1.getDimensions();
44  dimsStruct.simplify();
45  try
46  {
47  indexType nbElements = dimsStruct.getElementCount() * nbFields;
48  elements = new ArrayOf[nbElements];
49  }
50  catch (std::bad_alloc &e)
51  {
52  e.what();
53  Error(eval, ERROR_MEMORY_ALLOCATION);
54  }
55  try
56  {
57  ArrayOfVector *v = new ArrayOfVector[nbFields];
58  indexType S = 0;
59  for (size_t k = 0; k < nbFields; k++)
60  {
61  v[k] = param1.getFieldAsList(fieldnames[k]);
62  S = v[k].size();
63  }
64  size_t l = 0;
65  for (size_t q = 0; q < S; q++)
66  {
67  for (size_t k = 0; k < nbFields; k++)
68  {
69  elements[l] = v[k][q];
70  l++;
71  }
72  }
73  delete[] v;
74  }
75  catch (std::bad_alloc &e)
76  {
77  delete[] elements;
78  e.what();
79  Error(eval, ERROR_MEMORY_ALLOCATION);
80  }
81  Dimensions dims;
82  dims[0] = nbFields;
83  for (size_t k = 0; k < dimsStruct.getLength(); k++)
84  {
85  dims[k + 1] = dimsStruct[k];
86  }
87  dims.simplify();
88  ret.push_back(ArrayOf(NLS_CELL_ARRAY, dims, elements));
89  return ret;
90 }
