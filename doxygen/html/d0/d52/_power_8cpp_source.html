24  ArrayOf Power(ArrayOf A, ArrayOf B) throw(Exception)
25  {
26  if (A.isEmpty() || B.isEmpty())
27  {
28  return ArrayOf::emptyConstructor();
29  }
30  if (A.isScalar() && B.isScalar())
31  {
32  return DotPower(A, B);
33  }
34  // Check for A & B numeric
35  CheckNumeric(A, B, "^");
36  // Test for 2D on both A & B
37  if (!A.is2D() || !B.is2D())
38  {
39  throw Exception(_W("Cannot apply exponential operation to N-Dimensional arrays."));
40  }
41  // Test the types
42  TypeCheck(A, B, true);
43  // Both arguments must be square
44  if ((A.getDimensionLength(0) != A.getDimensionLength(1)) ||
45  (B.getDimensionLength(0) != B.getDimensionLength(1)))
46  {
47  throw Exception(_W("Power (^) operator can only be applied to scalar and square arguments."));
48  }
49  throw Exception(_W("Power (^) currently not implemented in Nelson."));
50  }
51 }
