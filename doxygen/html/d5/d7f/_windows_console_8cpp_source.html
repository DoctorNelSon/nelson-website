139 std::wstring WindowsConsole::getTextLine(std::wstring prompt, bool bIsInput)
140 {
141  std::wstring cmdline = L"";
142  if (prompt != L"")
143  {
144  lineObj.putCharacter(L'\n', LineManager::STANDARD_INPUT);
145  diary.writeMessage(L"\n");
146  }
147  lineObj.newLine();
148  lineObj.setCurrentPrompt(prompt);
149  lineObj.displayPrompt();
150  if (prompt != L"")
151  {
152  diary.writeMessage(prompt);
153  }
154  for (;;)
155  {
156  bool bIsAction = false;
157  wchar_t cur_char = getCharacter(bIsAction);
158  if ((cur_char == CR_1) || (cur_char == CR_2))
159  {
160  cmdline = lineObj.getCurrentLine();
161  lineObj.putCharacter(L'\n', LineManager::STANDARD_INPUT);
162  diary.writeMessage(cmdline + L"\n");
163  if (!bIsInput)
164  {
165  Nelson::History::addLine(cmdline);
166  }
167  if (bIsInput)
168  {
169  Nelson::History::setToken(L"");
170  if (boost::algorithm::ends_with(cmdline, L"\n"))
171  {
172  cmdline.pop_back();
173  }
174  }
175  if (cmdline.empty())
176  {
177  cmdline = L"\n";
178  }
179  return cmdline;
180  }
181  else if (bIsAction)
182  {
183  if (bIsInput)
184  {
185  Nelson::History::setToken(L"");
186  if (boost::algorithm::ends_with(cmdline, L"\n"))
187  {
188  cmdline.pop_back();
189  }
190  }
191  return cmdline;
192  }
193  else
194  {
195  lineObj.putCharacter(cur_char, LineManager::STANDARD_INPUT);
196  lineObj.addCharacterCurrentLine(cur_char);
197  std::wstring curline = lineObj.getCurrentLine();
198  if (!bIsInput)
199  {
200  Nelson::History::setToken(curline);
201  }
202  }
203  }
204  return cmdline;
205 }

279 wchar_t WindowsConsole::keysEventsFilter(INPUT_RECORD irBuffer, bool &bIsChar, bool &bIsAction)
280 {
281  DWORD n = 0;
282  wchar_t ch = 0;
283  bIsChar = false;
284  bIsAction = false;
285  Win32InputStream = GetStdHandle(STD_INPUT_HANDLE);
286  SetConsoleMode(Win32InputStream, ENABLE_PROCESSED_INPUT | ENABLE_QUICK_EDIT_MODE | ENABLE_INSERT_MODE | ENABLE_EXTENDED_FLAGS);
287  if (irBuffer.Event.KeyEvent.bKeyDown)
288  {
289  if (irBuffer.Event.KeyEvent.dwControlKeyState)
290  {
291  if (isCTRLPressed(irBuffer))
292  {
293  wchar_t c = actionControlKey();
294  if (c)
295  {
296  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
297  bIsChar = true;
298  return c;
299  }
300  else
301  {
302  if (irBuffer.Event.KeyEvent.uChar.UnicodeChar != L'\0')
303  {
304  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
305  c = irBuffer.Event.KeyEvent.uChar.UnicodeChar;
306  if ((c > 0) && !iswcntrl(c))
307  {
308  bIsChar = true;
309  return c;
310  }
311  }
312  else
313  {
314  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
315  }
316  }
317  bIsChar = false;
318  return 0;
319  }
320  if (isALTPressed(irBuffer))
321  {
322  if (irBuffer.Event.KeyEvent.uChar.AsciiChar != '\0')
323  {
324  ::ReadConsoleW(Win32InputStream, &ch, 1, &n, NULL);
325  bIsChar = true;
326  return ch;
327  }
328  else
329  {
330  WORD vk = 0;
331  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
332  vk = irBuffer.Event.KeyEvent.wVirtualKeyCode;
333  switch (vk)
334  {
335  case VK_F4:
336  AltF4_Command();
337  bIsAction = true;
338  return 0;
339  default:
340  break;
341  }
342  }
343  }
344  }
345  if (irBuffer.Event.KeyEvent.uChar.UnicodeChar != L'\0')
346  {
347  wchar_t _ch[2];
348  ::ReadConsoleW(Win32InputStream, &_ch, 1, &n, NULL);
349  ch = _ch[0];
350  switch (ch)
351  {
352  case VK_TAB:
353  break;
354  case VK_BACK:
355  lineObj.deletePreviousChar();
356  break;
357  default:
358  {
359  if (!iswcntrl(ch) || (ch == CR_1) || (ch == CR_2))
360  {
361  bIsChar = true;
362  return ch;
363  }
364  }
365  break;
366  }
367  }
368  else
369  {
370  WORD vk = 0;
371  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
372  vk = irBuffer.Event.KeyEvent.wVirtualKeyCode;
373  switch (vk)
374  {
375  case VK_F1:
376  case VK_HELP:
377  Help_Command();
378  bIsAction = true;
379  return 0;
380  break;
381  case VK_F2:
382  clearTerminal();
383  lineObj.displayPrompt();
384  break;
385  case VK_LEFT:
386  lineObj.moveBackSingleChar();
387  break;
388  case VK_RIGHT:
389  lineObj.moveForwardSingleChar();
390  break;
391  case VK_UP:
392  lineObj.moveBackHistory();
393  break;
394  case VK_DOWN:
395  lineObj.moveForwardHistory();
396  break;
397  case VK_DELETE:
398  {
399  lineObj.deleteCurrentChar();
400  std::wstring curline = lineObj.getCurrentLine();
401  Nelson::History::setToken(curline);
402  }
403  break;
404  case VK_HOME:
405  lineObj.moveBeginningLine();
406  break;
407  case VK_END:
408  lineObj.moveEndLine();
409  break;
410  case VK_PAUSE:
411  Pause_Command();
412  bIsAction = true;
413  return 0;
414  default:
415  break;
416  }
417  }
418  }
419  else
420  {
421  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
422  }
423  return ch;
424 }

426 wchar_t WindowsConsole::getCharacter(bool &bIsAction)
427 {
428  wchar_t wch = L'\0';
429  INPUT_RECORD irBuffer;
430  DWORD n = 0;
431  bCONTROLC = false;
432  do
433  {
434  Win32InputStream = GetStdHandle(STD_INPUT_HANDLE);
435  SetConsoleMode(Win32InputStream, ENABLE_PROCESSED_INPUT | ENABLE_QUICK_EDIT_MODE | ENABLE_INSERT_MODE | ENABLE_EXTENDED_FLAGS);
436  if (bHaveHisOwnWindow)
437  {
438  this->setConsoleTitle(L"Nelson");
439  }
440  HWND hwnd = GetConsoleWindow();
441  HMENU hmenu = GetSystemMenu(hwnd, FALSE);
442  EnableMenuItem(hmenu, SC_CLOSE, MF_GRAYED);
443  bIsAction = false;
444  ::WaitForSingleObject(Win32InputStream, INFINITE);
445  ::PeekConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
446  switch (irBuffer.EventType)
447  {
448  case KEY_EVENT:
449  {
450  bool bIsChar = false;
451  wchar_t _wch = keysEventsFilter(irBuffer, bIsChar, bIsAction);
452  if (bIsChar || bIsAction)
453  {
454  return _wch;
455  }
456  }
457  break;
458  case MENU_EVENT:
459  {
460  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
461  }
462  break;
463  case MOUSE_EVENT:
464  case WINDOW_BUFFER_SIZE_EVENT:
465  case FOCUS_EVENT:
466  default:
467  ::ReadConsoleInputW(Win32InputStream, &irBuffer, 1, &n);
468  break;
469  }
470  }
471  while (TRUE && !bCONTROLC);
472  if (bCONTROLC)
473  {
474  bCONTROLC = false;
475  bIsAction = true;
476  wch = 0;
477  }
478  return wch;
479 }

481 wchar_t WindowsConsole::actionControlKey(void)
482 {
483  if (isCTRL_VKEY(L'X'))
484  {
485  // managed by the signal
486  GenerateConsoleCtrlEvent(CTRL_C_EVENT, GetCurrentProcessId());
487  return 0;
488  }
489  else if (isCTRL_VKEY(L'C'))
490  {
491  // managed by the signal
492  GenerateConsoleCtrlEvent(CTRL_C_EVENT, GetCurrentProcessId());
493  return 0;
494  }
495  else if (isCTRL_VKEY('A')) /* moves to the beginning of the line */
496  {
497  lineObj.moveBeginningLine();
498  }
499  else if (isCTRL_VKEY('B')) /* moves back a single character */
500  {
501  lineObj.moveBackSingleChar();
502  }
503  else if (isCTRL_VKEY('D')) /* deletes the current character */
504  {
505  lineObj.deleteCurrentChar();
506  std::wstring curline = lineObj.getCurrentLine();
507  Nelson::History::setToken(curline);
508  }
509  else if (isCTRL_VKEY('E')) /* moves to the end of the line */
510  {
511  lineObj.moveEndLine();
512  }
513  else if (isCTRL_VKEY('F')) /* moves forward a single character */
514  {
515  lineObj.moveForwardSingleChar();
516  }
517  else if (isCTRL_VKEY('H')) /* delete the previous character */
518  {
519  lineObj.deletePreviousChar();
520  }
521  else if (isCTRL_VKEY('K')) /* kills from current position to the end of line */
522  {
523  lineObj.killCurrentPositionToEndLine();
524  }
525  else if (isCTRL_VKEY('N')) /* moves forward through history */
526  {
527  lineObj.moveForwardHistory();
528  }
529  else if (isCTRL_VKEY('P')) /* moves back through history */
530  {
531  lineObj.moveBackHistory();
532  }
533  else if (isCTRL_VKEY('R') || isCTRL_VKEY('L')) /* redraw line in case it gets trashed */
534  {
535  lineObj.redrawLine();
536  }
537  else if (isCTRL_VKEY('U')) /* kills the entire line */
538  {
539  lineObj.clearCurrentLine();
540  }
541  else if (isCTRL_VKEY('V'))
542  {
543  pasteClipBoard();
544  }
545  else if (isCTRL_VKEY('W')) /* kills last word */
546  {
547  lineObj.killLastWord();
548  }
549  else if (isCTRL_VKEY(VK_TAB) || isCTRL_VKEY(VK_SPACE)) /* Completion */
550  {
551  }
552  else if (isCTRL_VKEY(VK_LEFT)) /* */
553  {
554  lineObj.moveBackSingleWord();
555  }
556  else if (isCTRL_VKEY(VK_RIGHT)) /* */
557  {
558  lineObj.moveForwardSingleWord();
559  }
560  return 0;
561 }

599 void WindowsConsole::pasteClipBoard(void)
600 {
601  if (IsClipboardFormatAvailable(CF_UNICODETEXT) || IsClipboardFormatAvailable(CF_TEXT))
602  {
603  HGLOBAL hGMem = nullptr;
604  int typeClipboard = 0;
605  if (IsClipboardFormatAvailable(CF_UNICODETEXT))
606  {
607  typeClipboard = CF_UNICODETEXT;
608  }
609  else
610  {
611  typeClipboard = CF_TEXT;
612  }
613  if (::OpenClipboard(NULL))
614  {
615  hGMem = ::GetClipboardData(typeClipboard);
616  if (hGMem)
617  {
618  std::wstring currentLine = lineObj.getCurrentLine();
619  std::wstring newLine = currentLine;
620  if (typeClipboard == CF_UNICODETEXT)
621  {
622  LPWSTR lpMem = (LPWSTR)::GlobalLock(hGMem);
623  if (lpMem)
624  {
625  newLine = newLine + std::wstring(lpMem);
626  }
627  }
628  else
629  {
630  LPSTR lpMem = (LPSTR)::GlobalLock(hGMem);
631  if (lpMem)
632  {
633  newLine = newLine + utf8_to_wstring(std::string(lpMem));
634  }
635  }
636  ::GlobalUnlock(hGMem);
637  lineObj.clearCurrentLine();
638  lineObj.copyLine(newLine);
639  }
640  ::CloseClipboard();
641  }
642  }
643 }

652 bool WindowsConsole::copyToClipBoard(std::wstring txt)
653 {
654  bool bRes = false;
655  size_t lentxt = txt.size();
656  if (lentxt > 0)
657  {
658  OpenClipboard(NULL);
659  EmptyClipboard();
660  HGLOBAL hglbCopy = GlobalAlloc(GMEM_MOVEABLE, ((lentxt + 1) * sizeof(wchar_t)));
661  if (hglbCopy)
662  {
663  wchar_t* lptstrCopy = (wchar_t*)GlobalLock(hglbCopy);
664  if (lptstrCopy)
665  {
666  memcpy(lptstrCopy, txt.c_str(), lentxt * sizeof(wchar_t));
667  lptstrCopy[lentxt] = (wchar_t)0; // null character
668  GlobalUnlock(hglbCopy);
669  SetClipboardData(CF_UNICODETEXT, hglbCopy);
670  bRes = true;
671  }
672  }
673  CloseClipboard();
674  }
675  return bRes;
676 }
