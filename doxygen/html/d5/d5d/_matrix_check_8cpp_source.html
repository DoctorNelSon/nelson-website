23  void CheckNumeric(ArrayOf &A, ArrayOf &B, std::string opname) throw(Exception)
24  {
25  bool Anumeric, Bnumeric;
26  Anumeric = !A.isReferenceType();
27  Bnumeric = !B.isReferenceType();
28  if (!(Anumeric && Bnumeric))
29  throw Exception(std::string(_("Cannot apply numeric operation ")) +
30  opname + std::string(_(" to reference types.")));

33  bool MatrixCheck(ArrayOf &A, ArrayOf &B, std::string opname) throw(Exception)
34  {
35  // Test for either a scalar (test 1)
36  if (A.isScalar() || B.isScalar())
37  {
38  return false;
39  }
40  // Test for A & B numeric
41  CheckNumeric(A, B, opname);
42  // Test for 2D
43  if (!A.is2D() || !B.is2D())
44  throw Exception(std::string(_("Cannot apply matrix operation ")) +
45  opname + std::string(_(" to N-Dimensional arrays.")));
46  // Test the types
47  TypeCheck(A, B, true);
48  return true;
49  }

51  void TypeCheck(ArrayOf &A, ArrayOf &B, bool isDivOrMatrix)
52  {
53  Class Aclass, Bclass, Cclass;
54  Aclass = A.getDataClass();
55  Bclass = B.getDataClass();
56  if ( (Aclass == Bclass) &&
57  ( (Aclass == NLS_LOGICAL) ||
58  (Aclass == NLS_UINT8) ||
59  (Aclass == NLS_INT8) ||
60  (Aclass == NLS_UINT16) ||
61  (Aclass == NLS_INT16) ||
62  (Aclass == NLS_UINT32) ||
63  (Aclass == NLS_INT32) ||
64  (Aclass == NLS_UINT64) ||
65  (Aclass == NLS_INT64) ||
66  (Aclass == NLS_SINGLE) ||
67  (Aclass == NLS_DOUBLE) ||
68  (Aclass == NLS_SCOMPLEX) ||
69  (Aclass == NLS_DCOMPLEX) ||
70  (Aclass == NLS_STRING)) )
71  {
72  return;
73  }
74  if ((Aclass < NLS_INT32) || (Aclass == NLS_STRING))
75  {
76  Aclass = NLS_INT32;
77  }
78  if ((Bclass < NLS_INT32) || (Bclass == NLS_STRING))
79  {
80  Bclass = NLS_INT32;
81  }
82  // Division or matrix operations do no allow integer
83  // data types. These must be promoted to doubles.
84  if (isDivOrMatrix && (Aclass < NLS_SINGLE))
85  {
86  Aclass = NLS_DOUBLE;
87  }
88  if (isDivOrMatrix && (Bclass < NLS_SINGLE))
89  {
90  Bclass = NLS_DOUBLE;
91  }
92  // An integer or double mixed with a complex is promoted to a dcomplex type
93  if ((Aclass == NLS_SCOMPLEX) && ((Bclass == NLS_DOUBLE) || (Bclass < NLS_SINGLE)))
94  {
95  Bclass = NLS_DCOMPLEX;
96  }
97  if ((Bclass == NLS_SCOMPLEX) && ((Aclass == NLS_DOUBLE) || (Aclass < NLS_SINGLE)))
98  {
99  Aclass = NLS_DCOMPLEX;
100  }
101  // The output class is now the dominant class remaining:
102  Cclass = (Aclass > Bclass) ? Aclass : Bclass;
103  A.promoteType(Cclass);
104  B.promoteType(Cclass);
105  }
