92  void ArrayOf::makeSparse()
93  {
94  if (!is2D())
95  {
96  throw Exception(_W("Cannot make n-dimensional arrays sparse."));
97  }
98  if (isEmpty())
99  {
100  dp = dp->putData(dp->dataClass, dp->dimensions, NULL, true, dp->fieldNames);
101  return;
102  }
103  if (isReferenceType() || isString())
104  {
105  throw Exception(_W("Cannot make strings or reference types sparse."));
106  }
107  if (isSparse())
108  {
109  return;
110  }
111  if ((dp->dataClass == NLS_DOUBLE) ||
112  (dp->dataClass == NLS_DCOMPLEX) ||
113  (dp->dataClass == NLS_LOGICAL))
114  {
115  ensureSingleOwner();
116  }
117  else
118  {
119  throw Exception(_W("Cannot make sparse."));
120  }
121 #ifdef __NLS_USE_SPARSE_EIGEN
122  dp = dp->putData(dp->dataClass, dp->dimensions,

124  true,
125  dp->fieldNames);

127  dp = dp->putData(dp->dataClass, dp->dimensions,
