75  {
76  const ArrayOf* rptr;
77  Dimensions dims;
78  sizeType i, j;
79  ArrayOf *qp = nullptr;
80  try
81  {
82  if (fNames.size() != values.size())
83  {
84  throw Exception(_W("Number of field names must match number of values in structure constructor."));
85  }
92  bool nonSingularFound = false;
93  for (i = 0; i < (sizeType)values.size(); i++)
94  {
99  if (values[i].dp->dataClass == NLS_CELL_ARRAY)
100  {
104  if (!values[i].isScalar())
105  {
106  if (!nonSingularFound)
107  {
108  nonSingularFound = true;
109  dims = values[i].dp->dimensions;
110  }
111  else if (!dims.equals(values[i].dp->dimensions))
112  {
113  throw Exception(_W("ArrayOf dimensions of non-scalar entries must agree in structure construction."));
114  }
115  }
116  }
117  }
121  if (!nonSingularFound)
122  {
123  dims.reset();
124  dims[0] = 1;
125  dims[1] = 1;
126  }
132  qp = (ArrayOf*)allocateArrayOf(NLS_STRUCT_ARRAY, dims.getElementCount(), fNames);
136  indexType length = dims.getElementCount();
137  indexType offset = 0;
138  for (j = 0; j<length; j++)
139  for (i = 0; i < (sizeType)fNames.size(); i++)
140  {
141  ArrayOf rval = values[i];
142  if (rval.isSparse())
143  {
144  throw Exception(_W("sparse arrays not supported for struct constructor."));
145  }
146  rptr = (const ArrayOf*)rval.dp->getData();
147  if (rval.dp->dataClass == NLS_CELL_ARRAY)
148  {
149  if (rval.isScalar())
150  {
151  qp[offset] = rptr[0];
152  }
153  else
154  {
155  qp[offset] = rptr[j];
156  }
157  }
158  else
159  {
160  qp[offset] = rval;
161  }
162  offset++;
163  }
164  return ArrayOf(NLS_STRUCT_ARRAY, dims, qp, false, fNames);
165  }
166  catch (Exception &e)
167  {
168  e.what();
169  ArrayOf* rp = (ArrayOf*)qp;
170  delete[] rp;
171  rp = nullptr;
172  qp = nullptr;
173  throw;
174  }
175  }

187  ArrayOf ArrayOf::getField(std::string fieldName)
188  {
189  // First make sure that we are a scalar value.
190  if (!isScalar())
191  {
192  throw Exception(_W("Cannot dereference a field of a multi-element structure array."));
193  }
194  if (isSparse())
195  {
196  throw Exception(_W("getField not supported for sparse arrays."));
197  }
198  // Then, find the field index.
199  int64 ndx = getFieldIndex(fieldName);
200  if (ndx < 0)
201  {
202  throw Exception(_("Reference to non-existent field") + " " + fieldName);
203  }
204  // Cast real part into a data pointer, and return a copy of what the caller asked for!
205  const ArrayOf* qp = (const ArrayOf*)dp->getData();
206  return qp[ndx];
207  }

351  dp->setStructTypeName(classtype);
