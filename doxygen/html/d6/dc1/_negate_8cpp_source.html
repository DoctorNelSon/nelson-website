24  {

31  ArrayOf Negate(ArrayOf A) throw(Exception)
32  {
33  ArrayOf C;
34  Class Aclass;
35  if (A.isReferenceType())
36  {
37  throw Exception(L"Cannot negate non-numeric types.");
38  }
39  Aclass = A.getDataClass();
40  if (A.isSingleString())
41  {
42  Aclass = NLS_DOUBLE;
43  }
44  else if (Aclass < NLS_INT32)
45  {
46  Aclass = NLS_INT32;
47  }
48  A.promoteType(Aclass);
49  C = ArrayOf(Aclass, A.getDimensions(), NULL);
50  void *Cp = C.allocateArrayOf(Aclass, A.getLength());
51  switch (Aclass)
52  {
53  case NLS_INT32:
54  negate<int32>(A.getLength(), (int32*)Cp, (int32*)A.getDataPointer());
55  break;
56  case NLS_SINGLE:
57  negate<float>(A.getLength(), (float*)Cp, (float*)A.getDataPointer());
58  break;
59  case NLS_DOUBLE:
60  negate<double>(A.getLength(), (double*)Cp, (double*)A.getDataPointer());
61  break;
62  case NLS_SCOMPLEX:
63  negate<float>(2 * A.getLength(), (float*)Cp, (float*)A.getDataPointer());
64  break;
65  case NLS_DCOMPLEX:
66  negate<double>(2 * A.getLength(), (double*)Cp, (double*)A.getDataPointer());
67  break;
68  }
69  C.setDataPointer(Cp);
70  return C;
71  }
72 }
