26 {
27  ArrayOfVector retval;

29  {
30  Error(eval, ERROR_WRONG_NUMBERS_OUTPUT_ARGS);
31  }
32  Class cl = NLS_DOUBLE;
33  if (argIn.size() == 0)
34  {
35  retval.push_back(Zeros(eval, cl));
36  }
37  else
38  {
39  sizeType nRhs = argIn.size();
40  bool bCheckClassName = true;
41  if ((int)nRhs - 2 >= 0)
42  {
43  ArrayOf Arg = argIn[argIn.size() - 2];
44  if (Arg.isSingleString())
45  {
46  std::wstring paramstr = Arg.getContentsAsWideString();
47  if (paramstr == L"like")
48  {
49  ArrayOf lastArg = argIn[argIn.size() - 1];
50  switch (lastArg.getDataClass())
51  {
52  case NLS_DOUBLE:
53  cl = NLS_DOUBLE;
54  break;
55  case NLS_SINGLE:
56  cl = NLS_SINGLE;
57  break;
58  default:
59  Error(eval, _W("\'single\' or \'double\' expected at last argument."));
60  break;
61  }
62  nRhs = nRhs - 2;
63  bCheckClassName = false;
64  }
65  else
66  {
67  Error(eval, _W("\'like\' expected at n - 2 argument."));
68  }
69  }
70  }
71  ArrayOf lastArg = argIn[argIn.size() - 1];
72  if (lastArg.isSingleString() && bCheckClassName)
73  {
74  std::wstring paramstr = lastArg.getContentsAsWideString();
75  if (paramstr == L"double")
76  {
77  cl = NLS_DOUBLE;
78  nRhs--;
79  }
80  else if (paramstr == L"single")
81  {
82  cl = NLS_SINGLE;
83  nRhs--;
84  }
85  else
86  {
87  Error(eval, _W("\'single\' or \'double\' expected at last argument."));
88  }
89  }
90  if (nRhs == 0)
91  {
92  retval.push_back(Zeros(eval, cl));
93  return retval;
94  }

96  for (sizeType k = 0; k < nRhs; k++)
97  {
98  ArrayOf param = argIn[k];
99  indexType idx = param.getContentAsScalarIndex();
100  dims[k] = idx;
101  }
102  if (dims.getLength() == 1)
103  {
104  dims[1] = dims[0];
105  }
106  retval.push_back(Zeros(eval, dims, cl));
107  }
108  return retval;
109 }
