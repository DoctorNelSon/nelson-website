135  wstringVector ArrayOf::getContentAsWideStringVector(bool bCheckVector) const
136  {
137  wstringVector res;
138  if (dp->dataClass == NLS_STRING)
139  {
140  if (is2D())
141  {
142  indexType rows = getDimensions().getRows();
143  indexType columns = getDimensions().getColumns();
144  for (indexType i = 0; i < rows; i++)
145  {
146  std::wstring str;
147  charType *buffer = nullptr;
148  const charType *qp = nullptr;
149  qp = (const charType*)dp->getData();
150  for (indexType j = 0; j < columns; j++)
151  {
152  size_t idx = i + j * rows;
153  str.push_back(qp[idx]);
154  }
155  res.push_back(str);
156  str.clear();
157  }
158  }
159  else
160  {
161  res.push_back(getContentsAsWideString());
162  }
163  }
164  else
165  {
166  if (dp->dataClass != NLS_CELL_ARRAY)
167  {
168  throw Exception(_W("A cell expected."));
169  }
170  else if (isEmpty())
171  {
172  // res.push_back(L"");
173  }
174  else if (isVector() || !bCheckVector)
175  {
176  ArrayOf *arg = (ArrayOf*)(getDataPointer());
177  indexType nbElements = getDimensions().getElementCount();
178  res.reserve(nbElements);
179  for (indexType k = 0; k < nbElements; k++)
180  {
181  if (arg[k].getDataClass() != NLS_STRING)
182  {
183  res.clear();
184  throw Exception(_W("A cell of string expected."));
185  }
186  else
187  {
188  res.push_back(arg[k].getContentsAsWideString());
189  }
190  }
191  }
192  else
193  {
194  throw Exception(_W("A vector expected."));
195  }
196  }
197  return res;
198  }
