29  ArrayOf ArrayOf::cellConstructor(ArrayOfMatrix& m)
30  {
31  indexType columnCount = 0, rowCount = 0;
32  ArrayOf* qp = nullptr;
33  try
34  {
35  ArrayOfMatrix::iterator i = m.begin();
36  while (i != m.end())
37  {
38  ArrayOfVector ptr = *i;
43  if (i == m.begin())
44  {
45  columnCount = ptr.size();
46  }
47  else
48  {
52  if (ptr.size() != columnCount)
53  {
54  throw Exception(_W("Cell definition must have same number of elements in each row"));
55  }
56  }
57  ++i;
58  }
64  rowCount = m.size();
65  Dimensions retDims(2);
66  retDims[0] = rowCount;
67  retDims[1] = columnCount;
71  qp = (ArrayOf *)allocateArrayOf(NLS_CELL_ARRAY, retDims.getElementCount());
72  ArrayOf *sp;
76  sp = qp;
77  i = m.begin();
78  while (i != m.end())
79  {
80  ArrayOfVector ptr = *i;
81  ArrayOf *cp= sp;
82  for (sizeType j = 0; j < (sizeType)ptr.size(); j++)
83  {
84  *cp = ptr[j];
85  cp += rowCount;
86  }
87  ++i;
88  sp++;
89  }
90  return ArrayOf(NLS_CELL_ARRAY, retDims, qp);
91  }
92  catch (Exception &e)
93  {
94  ArrayOf* rp = (ArrayOf*)qp;
95  delete[] rp;
96  rp = nullptr;
97  qp = nullptr;
98  e.what();
99  throw;
100  }
101  }

106  ArrayOf ArrayOf::getVectorContents(ArrayOf& indexing)
107  {
108  if (this->isCell())
109  {
110  throw Exception(_W("Attempt to apply contents-indexing to non-cell array object."));
111  }
112  if (indexing.isEmpty())
113  {
114  throw Exception(_W("Empty contents indexing is not defined."));
115  }
116  if (isSparse())
117  {
118  throw Exception(_W("getVectorContents not supported for sparse arrays."));
119  }
120  indexing.toOrdinalType();
121  //
122  // The output is the same size as the _index_, not the
123  // source variable (neat, huh?). But it inherits the
124  // type of the source variable.
125  //
126  // The index HAS to be a scalar for contents-based addressing
127  if (indexing.getLength() != 1)
128  {
129  throw Exception(_W("Content indexing must return a single value."));
130  }
131  constIndexPtr index_p = (constIndexPtr)indexing.dp->getData();
132  if (*index_p == 0)
133  {
134  throw Exception(_W("Index exceeds cell array dimensions"));
135  }
136  else
137  {
138  indexType ndx = *index_p - 1;
139  indexType bound = getLength();
140  if (ndx >= bound)
141  {
142  throw Exception(_W("Index exceeds cell array dimensions"));
143  }
144  const ArrayOf* srcPart = (const ArrayOf*)dp->getData();
145  // Make a source of whatever is in that index, and return it.
146  return srcPart[ndx];
147  }
148  }

153  ArrayOf ArrayOf::getNDimContents(ArrayOfVector& indexing)
154  {
155  if (!this->isCell())
156  {
157  throw Exception(_W("Attempt to apply contents-indexing to non-cell array object."));
158  }
159  if (isSparse())
160  {
161  throw Exception(_W("getNDimContents not supported for sparse arrays."));
162  }
163  indexType L = indexing.size();
164  Dimensions outPos(L);
165  // Convert each of the indexing variables into an ordinal type.
166  // We don't catch any exceptions - let them propogate up the
167  // call chain.
168  // The index HAS to be a scalar for contents-based addressing.
169  for (indexType i = 0; i < L; i++)
170  {
171  indexing[i].toOrdinalType();
172  if (indexing[i].getLength() != 1)
173  {
174  throw Exception(_W("Content indexing must return a single value."));
175  }
176  constIndexPtr sp = (constIndexPtr)indexing[i].dp->getData();
177  outPos[i] = *sp - 1;
178  }
179  indexType j = dp->dimensions.mapPoint(outPos);
180  const ArrayOf* qp = (const ArrayOf*)dp->getData();
181  return qp[j];
182  }

187  ArrayOfVector ArrayOf::getVectorContentsAsList(ArrayOf& index)
188  {
189  ArrayOfVector m;
190  if (!this->isCell())
191  {
192  throw Exception(_W("Attempt to apply contents-indexing to non cell-array object."));
193  }
194  if (isSparse())
195  {
196  throw Exception(_W("getVectorContentsAsList not supported for sparse arrays."));
197  }
198  if (index.isEmpty())
199  {
200  return ArrayOfVector();
201  }
202  index.toOrdinalType();
203  // Get the maximum index
204  indexType max_index = index.getMaxAsIndex();
205  // Get our length
206  indexType bound = getLength();
207  if (max_index > bound)
208  {
209  throw Exception(_W("ArrayOf index exceeds bounds of cell-array"));
210  }
211  // Get the length of the index object
212  indexType index_length = index.getLength();
213  // Get a pointer to the index data set
214  constIndexPtr index_p = (constIndexPtr)index.dp->getData();
215  // Get a pointer to our data
216  const ArrayOf* qp = (const ArrayOf*)dp->getData();
217  // Now we copy data from dp to m
218  for (indexType i = 0; i<index_length; i++)
219  {
220  m.push_back(qp[index_p[i] - 1]);
221  }
222  return m;
223  }

228  ArrayOfVector ArrayOf::getNDimContentsAsList(ArrayOfVector& index)
229  {
230  if (!this->isCell())
231  {
232  throw Exception(_W("Attempt to apply contents-indexing to non cell-array object."));
233  }
234  if (isSparse())
235  {
236  throw Exception(_W("getNDimContentsAsList not supported for sparse arrays."));
237  }
238  // Store the return value here
239  ArrayOfVector m;
240  // Get the number of indexing dimensions
241  indexType L = index.size();
242  // Setup the data pointers
243  Dimensions outDims(L);
244  indexType i;
245  for (i = 0; i<L; i++)
246  {
247  index[i].toOrdinalType();
248  }
249  constIndexPtr* indx = new_with_exception<constIndexPtr>(L);
250  for (i = 0; i<L; i++)
251  {
252  outDims[i] = (index[i].getLength());
253  indx[i] = (constIndexPtr)index[i].dp->getData();
254  }
255  Dimensions argPointer(L);
256  Dimensions currentIndex(L);
257  const ArrayOf *qp = (const ArrayOf*)dp->getData();
258  indexType srcindex = 0;

260  {
261  for (indexType i = 0; i<L; i++)
262  {
263  currentIndex[i] = indx[i][argPointer[i]] - 1;
264  }
265  srcindex = dp->dimensions.mapPoint(currentIndex);
266  m.push_back(qp[srcindex]);
267  argPointer.incrementModulo(outDims, 0);
268  }

353  void ArrayOf::setVectorContentsAsList(ArrayOf& index, ArrayOfVector& data)
354  {
355  if (isSparse())
356  {
357  throw Exception(_W("setVectorContentsAsList not supported for sparse arrays."));
358  }
359  promoteType(NLS_CELL_ARRAY);

362  {
363  throw Exception(_W("Not enough right hand side values to satisy left hand side expression."));
364  }

367  // Resize us as necessary.

390  void ArrayOf::setNDimContentsAsList(ArrayOfVector& index, ArrayOfVector& data)
391  {
392  if (isSparse())
393  {
394  throw Exception(_W("setNDimContentsAsList not supported for sparse arrays."));
395  }
396  promoteType(NLS_CELL_ARRAY);
397  indexType L = index.size();
398  // Convert the indexing variables into an ordinal type.
399  indexType i;
400  for (i = 0; i<L; i++)
401  {
402  index[i].toOrdinalType();
403  }
404  // Set up data pointers
405  constIndexPtr* indx = new_with_exception<constIndexPtr>(L);
406  try
407  {
408  Dimensions a(L);
409  // First, we compute the maximum along each dimension.
410  // We also get pointers to each of the index pointers.
411  for (i = 0; i<L; i++)
412  {
413  a[i] = index[i].getMaxAsIndex();
414  indx[i] = (constIndexPtr)index[i].dp->getData();
415  }
416  // Next, we compute the number of entries in each component.
417  Dimensions argLengths(L);
418  Dimensions argPointer(L);
419  indexType dataCount = 1;
420  for (i = 0; i<L; i++)
421  {
422  argLengths[i] = index[i].getLength();
423  dataCount *= argLengths[i];
424  }
425  if ((int)data.size() < dataCount)
426  {
427  throw Exception(_W("Not enough right hand side values to satisfy left hand side expression"));
428  }
429  // Resize us as necessary
430  resize(a);
431  // Get a writable data pointer
432  ArrayOf *qp = (ArrayOf*)getReadWriteDataPointer();
433  // Now, we copy data from dp to our real part,
434  // computing indices along the way.
435  Dimensions currentIndex(dp->dimensions.getLength());
436  indexType j;
437  while (argPointer.inside(argLengths))
438  {
439  for (i = 0; i<L; i++)
440  {
441  currentIndex[i] = (indexType)indx[i][argPointer[i]] - 1;
442  }
443  j = dp->dimensions.mapPoint(currentIndex);
444  qp[j] = data.front();
445  // data.pop_front();
446  data.erase(data.begin());
447  argPointer.incrementModulo(argLengths, 0);
448  }
449  delete[] indx;
450  indx = nullptr;
451  dp->dimensions.simplify();
452  }
453  catch (Exception &e)
454  {
455  delete[] indx;
456  indx = nullptr;
457  e.what();
458  throw;
459  }
460  }
