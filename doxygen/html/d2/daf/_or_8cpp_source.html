23  void boolean_or(indexType N, logical* C, const logical *A, int Astride,
24  const logical *B, int Bstride)
25  {
26  indexType m, p;
27  m = 0;
28  p = 0;
29  for (indexType i = 0; i<N; i++)
30  {
31  C[i] = A[m] || B[p];
32  m += Astride;
33  p += Bstride;
34  }
35  }
