55  else if ((arg == L"b") || (arg == L"ieee-be"))

58  }

62  }
63  else

66  }
67  skipSize = (size_t)param4.getContentAsScalarIndex();
68  if (param3.isSingleString())
69  {
70  std::wstring precisionStr = param3.getContentsAsWideString();
71  bool bOK = false;
72  classDest = precisionFromString(precisionStr, bOK);
73  if (!bOK)
74  {
75  Error(eval, _W("Wrong value for #3 argument: not supported precision."));
76  }
77  }
78  else
79  {
80  Error(eval, ERROR_WRONG_ARGUMENT_3_TYPE_STRING_EXPECTED);
81  }
82  if (param1.isDoubleType())
83  {
84  if (!param2.isNumeric())
85  {
86  Error(eval, ERROR_WRONG_ARGUMENT_2_TYPE_NUMERIC_EXPECTED);
87  }
88  bool bSizeIs2D = param2.is2D() && !param2.isScalar();
89  int64 isize = 0;
90  int64 im = 0;
91  int64 in = 0;
92  if (bSizeIs2D)
93  {
94  // [m, n]
95  // [m, Inf]

97  double *dValues = (double*)param2.getReadWriteDataPointer();

99  double n = dValues[1];
100  if (std::isinf(m))

103  }
104  im = (int64)m;
105  if (std::isinf(n))

107  if (n > 0)
108  {
109  isize = -1;
110  }
111  else
112  {
113  Error(eval, ERROR_WRONG_ARGUMENT_2_INVALID_VECTOR_SIZE);
114  }
115  }
116  else

120  }
121  }
122  else

129  isize = -1;
130  }
131  else
132  {
133  Error(eval, ERROR_WRONG_ARGUMENT_2_INVALID_VECTOR_SIZE);
134  }
135  }
136  else
137  {
138  isize = (int64)dsize;
139  }
140  }
141  FilesManager *fm = (FilesManager *)(eval->FileManager);
142  int32 iValue = (int32)param1.getContentsAsDoubleScalar();
143  if (fm == nullptr)

145  Error(eval, _W("Problem with file manager."));
146  }

149  File *f = fm->getFile(iValue);

152  if (sizeReallyRead != -1)
153  {
154  if (bSizeIs2D)
155  {
156  Dimensions dim;
157  if (isize == -1)

159  // n is inf

161  if (sizeReallyRead % im)
162  {
163  in++;
164  }
165  Dimensions dimL((indexType)im, (indexType)in);
166  dim = dimL;
167  }
168  else
169  {
170  Dimensions dimL((indexType)im, (indexType)in);
171  dim = dimL;
172  }
173  if (sizeReallyRead == im * in)
174  {
175  toRead.reshape(dim);
176  retval.push_back(toRead);
177  }
178  else
179  {
180  void *ptr = ArrayOf::allocateArrayOf(toRead.getDataClass(), (indexType)(im * in));
181  memcpy(ptr, toRead.getReadWriteDataPointer(), toRead.getByteSize());
182  ArrayOf Resized = ArrayOf(toRead.getDataClass(), dim, ptr);
183  retval.push_back(Resized);
184  }
185  }
186  else
187  {
188  retval.push_back(toRead);
189  }
190  if (nLhs > 1)
191  {
192  retval.push_back(ArrayOf::doubleConstructor(sizeReallyRead));
193  }
194  }
195  else
196  {
197  Error(eval, _W("Problem to read data."));
198  }
199  }
200  else
201  {
202  Error(eval, _W("Invalid file identifier."));
203  }
204  }
205  else
206  {
207  Error(eval, _W("Invalid file identifier."));
208  }
209  return retval;
210 }
