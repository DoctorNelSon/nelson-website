47  static ArrayOf OverloadBinaryOperator(Evaluator *eval, ArrayOf a, ArrayOf b, std::string functionName, bool bRaiseError, bool &bSuccess, std::string forcedFunctionName)
48  {
49  FunctionDef *funcDef = nullptr;
50  std::string forcedName;
51  std::string OverloadName;
52  bSuccess = OverloadBinaryOperatorFindFunction(eval, a, b, functionName, forcedFunctionName, &funcDef, OverloadName);
53  if (!bSuccess)
54  {
55  forcedName = ClassName(a) + std::string("_") + functionName + std::string("_") + NLS_GENERIC_STR;
56  bSuccess = OverloadBinaryOperatorFindFunction(eval, a, b, functionName, forcedName, &funcDef, forcedName);
57  }
58  if (!bSuccess)
59  {
60  forcedName = NLS_GENERIC_STR + std::string("_") + functionName + std::string("_") + ClassName(b);
61  bSuccess = OverloadBinaryOperatorFindFunction(eval, a, b, functionName, forcedName, &funcDef, forcedName);
62  }
63  if (!bSuccess)
64  {
65  forcedName = NLS_GENERIC_STR + std::string("_") + functionName + std::string("_") + NLS_GENERIC_STR;
66  bSuccess = OverloadBinaryOperatorFindFunction(eval, a, b, functionName, forcedName, &funcDef, forcedName);
67  }
68  if (!bSuccess)
69  {
70  if (bRaiseError)
71  {
72  throw Exception(std::string("function ") + OverloadName + " undefined.");
73  }
74  return ArrayOf::emptyConstructor();
75  }
76  ArrayOfVector argsIn;
77  argsIn.push_back(a);
78  argsIn.push_back(b);
79  int nargout = 1;
80  ArrayOfVector res = funcDef->evaluateFunction(eval, argsIn, nargout);
81  if (res.size() != 1)
82  {
83  if (bRaiseError)
84  {
85  throw Exception(std::string("function ") + OverloadName + " only one output argument expected.");
86  }
87  bSuccess = false;
88  return ArrayOf::emptyConstructor();
89  }
90  return res[0];
91  }
