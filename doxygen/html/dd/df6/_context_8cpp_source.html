116  {

126  mapName = varName;

240  void Context::addPersistentVariable(std::string var)
241  {
242  // Delete local variables with this name
243  scopestack.back()->deleteVariable(var);
244  // Delete global variables with this name
245  scopestack.front()->deleteVariable(var);
246  scopestack.back()->addPersistentVariablePointer(var);
247  ArrayOf v;
248  if (!scopestack.front()->lookupVariable(scopestack.back()->getMangledName(var), v))
249  {
250  scopestack.front()->insertVariable(scopestack.back()->getMangledName(var), ArrayOf::emptyConstructor());
251  }
252  }
253  //=============================================================================
254  void Context::addGlobalVariable(std::string var)
255  {
256  // Delete local variables with this name
257  scopestack.back()->deleteVariable(var);
258  // Delete global persistent variables with this name
259  scopestack.front()->deleteVariable(scopestack.back()->getMangledName(var));
260  // Add a point in the local scope to the global variable
261  scopestack.back()->addGlobalVariablePointer(var);
262  // Make sure the variable exists
263  ArrayOf v;
264  if (!scopestack.front()->lookupVariable(var, v))
265  {
266  scopestack.front()->insertVariable(var, ArrayOf::emptyConstructor());
267  }
268  }

281  scopestack.back()->deletePersistentVariablePointer(var);
282  return;

284  scopestack.back()->deleteVariable(var);
