44 ArrayOfVector Nelson::RandomGateway::rngBuiltin(Evaluator* eval, int nLhs, const ArrayOfVector& argIn)
45 {
46  ArrayOfVector retval;
47  if (nLhs > 1)
48  {
49  Error(eval, ERROR_WRONG_NUMBERS_OUTPUT_ARGS);
50  }
51  switch (argIn.size())
52  {
53  case 0:
54  {
55  if (eval->RandomEngine == nullptr)
56  {
57  Error(eval, _W("random engine not initialized."));
58  }
59  retval.push_back(backupCurrentRng(eval));
60  }
61  break;
62  case 1:
63  {
64  ArrayOf arg1 = argIn[0];
65  if (arg1.isSingleString())
66  {
67  std::wstring param = arg1.getContentsAsWideString();
68  if (!((param == L"default") || (param == L"shuffle") || (param == L"enginelist")))
69  {
70  Error(eval, _W("'default', 'shuffle' or 'enginelist' expected."));
71  }
72  ArrayOf backupCurrentRngStruct;
73  if (nLhs == 1)
74  {
75  backupCurrentRngStruct = backupCurrentRng(eval);
76  }
77  if (param == L"default")
78  {
79  RngSetDefault(eval);
80  if (nLhs == 1)
81  {
82  retval.push_back(backupCurrentRngStruct);
83  }
84  }
85  else if (param == L"shuffle")
86  {
87  RngShuffle(eval);
88  if (nLhs == 1)
89  {
90  retval.push_back(backupCurrentRngStruct);
91  }
92  }
93  else if (param == L"enginelist")
94  {
95  wstringVector enginelist = getSupportedRngEngineName();
96  retval.push_back(ToCellStringAsColumn(enginelist));
97  }
98  }
99  else if (arg1.isNumeric())
100  {
101  double s = arg1.getContentsAsDoubleScalar();
102  ArrayOf backupCurrentRngStruct;
103  if (nLhs == 1)
104  {
105  backupCurrentRngStruct = backupCurrentRng(eval);
106  }
107  RngSetSeed(eval, s);
108  if (nLhs == 1)
109  {
110  retval.push_back(backupCurrentRngStruct);
111  }
112  }
113  else if (arg1.isStruct())
114  {
115  ArrayOfVector elements = splitRngStruct(eval, arg1);
116  std::wstring genname = elements[0].getContentsAsWideString();
117  double s = elements[1].getContentsAsDoubleScalar();
118  RngSetEngine(eval, s, genname);
119  RngSetState(eval, elements[2]);
120  }
121  else
122  {
123  Error(eval, ERROR_WRONG_ARGUMENT_1_TYPE);
124  }
125  }
126  break;
127  case 2:
128  {
129  ArrayOf arg1 = argIn[0];
130  ArrayOf arg2 = argIn[1];
131  if (arg1.isNumeric() && arg2.isSingleString())
132  {
133  double s = arg1.getContentsAsDoubleScalar();
134  std::wstring genname = arg2.getContentsAsWideString();
135  if (!isRngType(genname))
136  {
137  Error(eval, _W("A valid generator name expected."));
138  }
139  ArrayOf backupCurrentRngStruct;
140  if (nLhs == 1)
141  {
142  backupCurrentRngStruct = backupCurrentRng(eval);
143  }
144  RngSetEngine(eval, s, genname);
145  if (nLhs == 1)
146  {
147  retval.push_back(backupCurrentRngStruct);
148  }
149  }
150  else if (arg1.isSingleString() && arg2.isSingleString())
151  {
152  std::wstring param = arg1.getContentsAsWideString();
153  if (param != L"shuffle")
154  {
155  Error(eval, _W("'shuffle' expected."));
156  }
157  std::wstring genname = arg2.getContentsAsWideString();
158  if (!isRngType(genname))
159  {
160  Error(eval, _W("A valid generator name expected."));
161  }
162  ArrayOf backupCurrentRngStruct;
163  if (nLhs == 1)
164  {
165  backupCurrentRngStruct = backupCurrentRng(eval);
166  }
167  RngSetEngine(eval, 0, genname);
168  RngShuffle(eval);
169  if (nLhs == 1)
170  {
171  retval.push_back(backupCurrentRngStruct);
172  }
173  }
174  else
175  {
176  Error(eval, ERROR_WRONG_ARGUMENTS_TYPE);
177  }
178  }
179  break;
180  default:
181  {
182  Error(eval, ERROR_WRONG_NUMBERS_INPUT_ARGS);
183  }
184  break;
185  }
186  return retval;
187 }

189 static ArrayOf backupCurrentRng(Evaluator* eval)
190 {
191  ArrayOf typeOfRng = ArrayOf::stringConstructor(RngGetType(eval));
192  ArrayOf seedOfRng = RngGetSeed(eval);
193  ArrayOf stateOfRng = RngGetState(eval);
194  ArrayOf structRng;
195  ArrayOfVector fieldvalues;
196  stringVector fieldnames;
197  fieldvalues.push_back(typeOfRng);
198  fieldvalues.push_back(seedOfRng);
199  fieldvalues.push_back(stateOfRng);
200  fieldnames.push_back("Type");
201  fieldnames.push_back("Seed");
202  fieldnames.push_back("State");
203  ArrayOf retStruct = ArrayOf::structConstructor(fieldnames, fieldvalues);
204  return retStruct;
205 }

207 static ArrayOfVector splitRngStruct(Evaluator* eval, ArrayOf structRng)
208 {
209  ArrayOfVector elements;
210  stringVector fieldnames = structRng.getFieldNames();
211  if (fieldnames.size() == 3)
212  {
213  if ((fieldnames[0] == "Type") && (fieldnames[1] == "Seed") && (fieldnames[2] == "State"))
214  {
215  ArrayOf nType = structRng.getField("Type");
216  ArrayOf nSeed = structRng.getField("Seed");
217  ArrayOf nState = structRng.getField("State");
218  elements.push_back(nType);
219  elements.push_back(nSeed);
220  elements.push_back(nState);
221  return elements;
222  }
223  }
224  Error(eval, _W("Must contain generator settings captured previously."));
225  return elements;
226 }
