25  static ArrayOf double_times(ArrayOf a, ArrayOf b)
26  {
27  if (a.isScalar() && b.isScalar())
28  {
29  // s .* s
30  return ArrayOf::doubleConstructor(a.getContentsAsDoubleScalar() * b.getContentsAsDoubleScalar());
31  }
32  else
33  {
34  if (a.isScalar() || b.isScalar())
35  {
36  // mxn .* s
37  // s .* mxn
38  Dimensions dimsC;
39  void *Cp = nullptr;
40  indexType Clen;
41  if (a.isScalar())
42  {
43  double da = a.getContentsAsDoubleScalar();
44  dimsC = b.getDimensions();
45  Clen = dimsC.getElementCount();
46  Cp = new_with_exception<double>(Clen);
47  Eigen::Map<Eigen::MatrixXd> matC((double*)Cp, 1, Clen);
48  Eigen::Map<Eigen::MatrixXd> matB((double*)b.getDataPointer(), 1, Clen);
49  matC = da * matB.array();
50  }
51  else
52  {
53  double db = b.getContentsAsDoubleScalar();
54  dimsC = a.getDimensions();
55  Clen = dimsC.getElementCount();
56  Cp = new_with_exception<double>(Clen);
57  Eigen::Map<Eigen::MatrixXd> matC((double*)Cp, 1, Clen);
58  Eigen::Map<Eigen::MatrixXd> matA((double*)a.getDataPointer(), 1, Clen);
59  matC = matA.array() * db;
60  }
61  return ArrayOf(NLS_DOUBLE, dimsC, Cp, false);
62  }
63  else
64  {
65  // mxn .* mxn
66  Dimensions dimsC = a.getDimensions();
67  if (a.isEmpty(true))
68  {
69  return ArrayOf::emptyConstructor(dimsC);
70  }
71  else
72  {
73  indexType Clen = dimsC.getElementCount();
74  void *Cp = new_with_exception<double>(Clen);
75  Eigen::Map<Eigen::MatrixXd> matC((double*)Cp, 1, Clen);
76  Eigen::Map<Eigen::MatrixXd> matA((double*)a.getDataPointer(), 1, Clen);
77  Eigen::Map<Eigen::MatrixXd> matB((double*)b.getDataPointer(), 1, Clen);
78  matC = matA.cwiseProduct(matB);
79  return ArrayOf(NLS_DOUBLE, dimsC, Cp, false);
80  }
81  }
82  }
83  return ArrayOf();
84  }

86  static ArrayOf dcomplex_times(ArrayOf a, ArrayOf b)
87  {
88  ArrayOf res;
89  a.promoteType(NLS_DCOMPLEX);
90  b.promoteType(NLS_DCOMPLEX);
91  if (a.isScalar() && b.isScalar())
92  {
93  // s .* s
94  void *Cp = new_with_exception<double>(2);
95  doublecomplex* Cz = reinterpret_cast<doublecomplex*>(Cp);
96  doublecomplex* Az = reinterpret_cast<doublecomplex*>((double*)a.getDataPointer());
97  doublecomplex cxa = Az[0];
98  doublecomplex* Bz = reinterpret_cast<doublecomplex*>((double*)b.getDataPointer());
99  doublecomplex cxb = Bz[0];
100  Cz[0] = cxa * cxb;
101  res =ArrayOf(NLS_DCOMPLEX, a.getDimensions(), Cp, false);
102  }
103  else
104  {
105  if (a.isScalar() || b.isScalar())
106  {
107  // mxn .* s
108  // s .* mxn
109  // s .* s
110  Dimensions dimsC;
111  void *Cp = nullptr;
112  if (a.isScalar())
113  {
114  dimsC = b.getDimensions();
115  Cp = new_with_exception<double>(2 * dimsC.getElementCount());
116  doublecomplex* Cz = reinterpret_cast<doublecomplex*>(Cp);
117  Eigen::Map<Eigen::MatrixXcd> matC(Cz, 1, dimsC.getElementCount());
118  doublecomplex* Az = reinterpret_cast<doublecomplex*>((double*)a.getDataPointer());
119  doublecomplex* Bz = reinterpret_cast<doublecomplex*>((double*)b.getDataPointer());
120  Eigen::Map<Eigen::MatrixXcd> matB(Bz, 1, dimsC.getElementCount());
121  matC = Az[0] * matB.array();
122  }
123  else
124  {
125  dimsC = a.getDimensions();
126  Cp = new_with_exception<double>(2 * dimsC.getElementCount());
127  doublecomplex* Cz = reinterpret_cast<doublecomplex*>(Cp);
128  Eigen::Map<Eigen::MatrixXcd> matC(Cz, 1, dimsC.getElementCount());
129  doublecomplex* Bz = reinterpret_cast<doublecomplex*>((double*)b.getDataPointer());
130  doublecomplex* Az = reinterpret_cast<doublecomplex*>((double*)a.getDataPointer());
131  Eigen::Map<Eigen::MatrixXcd> matA(Az, 1, dimsC.getElementCount());
132  matC = matA.array() * Bz[0];
133  }
134  res = ArrayOf(NLS_DCOMPLEX, dimsC, Cp, false);
135  }
136  else
137  {
138  // mxn .* mxn
139  Dimensions dimsC = a.getDimensions();
140  if (a.isEmpty(true))
141  {
142  return ArrayOf::emptyConstructor(dimsC);
143  }
144  else
145  {
146  void *Cp = new_with_exception<double>(2 * dimsC.getElementCount());
147  doublecomplex* Cz = reinterpret_cast<doublecomplex*>(Cp);
148  doublecomplex* Az = reinterpret_cast<doublecomplex*>((double*)a.getDataPointer());
149  Eigen::Map<Eigen::MatrixXcd> matA(Az, 1, dimsC.getElementCount());
150  doublecomplex* Bz = reinterpret_cast<doublecomplex*>((double*)b.getDataPointer());
151  Eigen::Map<Eigen::MatrixXcd> matB(Bz, 1, dimsC.getElementCount());
152  Eigen::Map<Eigen::MatrixXcd> matC(Cz, 1, dimsC.getElementCount());
153  matC = matA.cwiseProduct(matB);
154  res = ArrayOf(NLS_DCOMPLEX, dimsC, Cp, false);
155  }
156  }
157  }
158  if (res.allReal())
159  {
160  res.promoteType(NLS_DOUBLE);
161  }
162  return res;
163  }

165  ArrayOf double_times_double(ArrayOf a, ArrayOf b)
166  {
167  if (!a.isDoubleType() || !b.isDoubleType())
168  {
169  throw Exception(ERROR_WRONG_ARGUMENTS_TYPE_DOUBLE_EXPECTED);
170  }
171  if (a.isSparse() || b.isSparse())
172  {
173  throw Exception(ERROR_WRONG_ARGUMENTS_SIZE_FULL_MATRIX_EXPECTED);
174  }
175  if (!(SameSizeCheck(a.getDimensions(), b.getDimensions()) || a.isScalar() || b.isScalar()))
176  {
177  throw Exception(_W("Size mismatch on arguments to arithmetic operator ") + L"*");
178  }
179  if (a.isComplex() || b.isComplex())
180  {
181  return dcomplex_times(a, b);
182  }
183  return double_times(a, b);
184  }
