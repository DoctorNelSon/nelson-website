26  void equalsfuncreal(indexType N, logical* C, const T*A, int stride1, const T*B, int stride2)
27  {

29  {
30 #if defined(__NLS_WITH_OPENMP)

34  {

36  }
37  }

64  void equalsfunccomplex(indexType N, logical* C, const T*A, int stride1, const T*B, int stride2)
65  {
66  if ((stride1 == 1) && (stride2 == 1))
67  {
68 #if defined(__NLS_WITH_OPENMP)
69  #pragma omp parallel for
70 #endif
71  for (indexType i = 0; i < N; i++)
72  {
73  C[i] = ((A[2 * i] == B[2 * i]) && (A[2 * i + 1] == B[2 * i + 1])) ? 1 : 0;
74  }
75  }
76  else
