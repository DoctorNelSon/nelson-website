28 ArrayOfVector Nelson::FilesFoldersGateway::dirBuiltin(Evaluator* eval, int nLhs, const ArrayOfVector& argIn)
29 {
30  ArrayOfVector retval;
31  std::wstring wpath;
32  std::wstring woption;
33  bool bSubDirectories = false;
34  if (nLhs > 1)
35  {
36  Error(eval, ERROR_WRONG_NUMBERS_OUTPUT_ARGS);
37  }
38  switch (argIn.size())
39  {
40  case 0:
41  {
42  wpath = GetCurrentDirectory();
43  if (wpath == L"")
44  {
45  Error(eval, _W("Impossible to get current directory."));
46  }
47  }
48  break;
49  case 2:
50  {
51  if (!argIn[1].isSingleString())
52  {
53  Error(eval, ERROR_WRONG_ARGUMENT_2_TYPE_STRING_EXPECTED);
54  }
55  woption = argIn[1].getContentsAsWideString();
56  if (woption == L"-s")
57  {
58  bSubDirectories = true;
59  }
60  else
61  {
62  Error(eval, ERROR_WRONG_ARGUMENT_2_VALUE);
63  }
64  }
65  case 1:
66  {
67  if (!argIn[0].isSingleString())
68  {
69  Error(eval, ERROR_WRONG_ARGUMENT_1_TYPE_STRING_EXPECTED);
70  }
71  wpath = argIn[0].getContentsAsWideString();
72  }
73  break;
74  default:
75  Error(eval, ERROR_WRONG_NUMBERS_INPUT_ARGS);
76  break;
77  }
78  boost::container::vector<FileInfo> res = ListFiles(wpath, bSubDirectories);
79  if (nLhs == 0)
80  {
81  Interface *io = eval->getInterface();
82  if (res.empty())
83  {
84  std::wstring msg = std::wstring(L"\'") + wpath + std::wstring(L"\' ") + _W("Not a file or a directory.");
85  io->outputMessage(msg);
86  }
87  else
88  {
89  for (boost::container::vector<FileInfo>::iterator it = res.begin(); it != res.end(); ++it)
90  {
91  if (eval->GetInterruptPending())
92  {
93  break;
94  }
95  if (it->isDir())
96  {
97  if (it->getName() == L"." || it->getName() == L"..")
98  {
99  io->outputMessage(it->getName() + L"\n");
100  }
101  else
102  {
103  io->outputMessage(it->getName() + L"/" + L"\n");
104  }
105  }
106  else
107  {
108  io->outputMessage(it->getName() + L"\n");
109  }
110  }
111  }
112  }
113  else
114  {
115  stringVector fieldnames;
116  fieldnames.push_back("name");
117  fieldnames.push_back("date");
118  fieldnames.push_back("bytes");
119  fieldnames.push_back("isdir");
120  fieldnames.push_back("datenum");
121  Dimensions dims;
122  dims[0] = res.size();
123  dims[1] = 1;
124  if (res.empty())
125  {
126  retval.push_back(ArrayOf::emptyStructConstructor(fieldnames, dims));
127  }
128  else
129  {
130  ArrayOf *elements = (ArrayOf*)ArrayOf::allocateArrayOf(NLS_STRUCT_ARRAY, dims.getElementCount(),fieldnames);
131  ArrayOf st = ArrayOf(NLS_STRUCT_ARRAY, dims, elements, false, fieldnames);
132  ArrayOfVector names;
133  ArrayOfVector dates;
134  ArrayOfVector bytes;
135  ArrayOfVector isdirs;
136  ArrayOfVector datenums;
137  names.reserve(res.size());
138  dates.reserve(res.size());
139  bytes.reserve(res.size());
140  isdirs.reserve(res.size());
141  datenums.reserve(res.size());
142  for (boost::container::vector<FileInfo>::iterator it = res.begin(); it != res.end(); ++it)
143  {
144  names.push_back(ArrayOf::stringConstructor(it->getName()));
145  dates.push_back(ArrayOf::stringConstructor(it->getDate()));
146  double bytesval = it->getBytes();
147  if (bytesval == -1)
148  {
149  bytes.push_back(ArrayOf::emptyConstructor());
150  }
151  else
152  {
153  bytes.push_back(ArrayOf::doubleConstructor(bytesval));
154  }
155  isdirs.push_back(ArrayOf::logicalConstructor(it->isDir()));
156  double datenumval = it->getDatenum();
157  if (datenumval == -1)
158  {
159  datenums.push_back(ArrayOf::emptyConstructor());
160  }
161  else
162  {
163  datenums.push_back(ArrayOf::doubleConstructor(datenumval));
164  }
165  }
166  st.setFieldAsList("name", names);
167  st.setFieldAsList("date", dates);
168  st.setFieldAsList("bytes", bytes);
169  st.setFieldAsList("isdir", isdirs);
170  st.setFieldAsList("datenum", datenums);
171  retval.push_back(st);
172  }
173  }
174  return retval;
175 }
